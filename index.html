def bm25_score(query, document, corpus):
    """Okapi BM25 — ranqueamento de relevância"""
    score = 0.0
    doc_len = len(document)
    avg_len = sum(len(d) for d in corpus) / len(corpus)
    
    for term in query:
        tf = document.count(term)
        df = sum(1 for d in corpus if term in d)
        idf = log((N - df + 0.5) / (df + 0.5))
        
        numerator = tf * (k1 + 1)
        denominator = tf + k1 * (1 - b + b * doc_len / avg_len)
        score += idf * numerator / denominator
    
    return score  # → Relevância calculada
